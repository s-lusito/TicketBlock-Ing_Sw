\section{Introduzione}

% Slide 4: Problema
\begin{frame}{Il Problema}
\begin{block}{Problematiche del Ticketing Tradizionale}
\begin{itemize}
    \item \textbf{Contraffazione}: Biglietti falsi difficili da identificare
    \item \textbf{Bagarinaggio}: Rivendita speculativa a prezzi esorbitanti
    \item \textbf{Mancanza di trasparenza}: Storia del biglietto non tracciabile
    \item \textbf{Frodi}: Rischio di acquistare biglietti non validi
    \item \textbf{Mercato secondario}: Nessun controllo sulle rivendite
\end{itemize}
\end{block}
\end{frame}

% Slide 5: Soluzione
\begin{frame}{La Nostra Soluzione}
\begin{block}{TicketBlock: Biglietteria su Blockchain}
\begin{columns}
\column{0.5\textwidth}
\textbf{Caratteristiche Principali:}
\begin{itemize}
    \item NFT per ogni biglietto
    \item Smart contracts Ethereum
    \item Rivendita controllata
    \item Trasparenza totale
    \item Anti-bagarinaggio
\end{itemize}

\column{0.5\textwidth}
\textbf{Vantaggi:}
\begin{itemize}
    \item Autenticità garantita
    \item Tracciabilità completa
    \item Fee opzionale 10\%
    \item Limite 4 ticket/evento
    \item Sicurezza blockchain
\end{itemize}
\end{columns}
\end{block}
\end{frame}

% Slide 13: Casi d'Uso Principali
\begin{frame}{Casi d'Uso - Creazione Evento}
\begin{enumerate}
    \item Organizzatore seleziona venue disponibile
    \item Inserisce dettagli evento (nome, data, prezzi)
    \item Sistema valida disponibilità venue
    \item Sistema genera biglietti per tutti i posti
    \item Evento pubblicato con stato vendita
\end{enumerate}

\vspace{0.5cm}
\begin{center}
\textbf{Regole Business:}
\begin{itemize}
    \item Vendite iniziano almeno 3 giorni prima evento
    \item Vendite terminano 1 giorno prima evento
    \item Prezzi differenziati STANDARD/VIP
\end{itemize}
\end{center}
\end{frame}

% Slide 14: Casi d'Uso - Acquisto Biglietto
\begin{frame}{Casi d'Uso - Acquisto Biglietto}
\begin{columns}
\column{0.6\textwidth}
\textbf{Flusso Acquisto:}
\begin{enumerate}
    \item Utente seleziona biglietti (max 4)
    \item Sceglie opzione resellable (+10\%)
    \item Inserisce dati pagamento
    \item Sistema processa pagamento
    \item Mint/Transfer NFT blockchain
    \item Conferma acquisto
\end{enumerate}

\column{0.4\textwidth}
\textbf{Blockchain:}
\begin{itemize}
    \item Primo acquisto: Mint NFT
    \item Rivendita: Transfer NFT
    \item Gas ottimizzato
    \item Transazione tracciata
\end{itemize}
\end{columns}
\end{frame}
